# reactive

返回对象的响应式副本

<template>
  <div class="home">
  	<!--模板中使用-->  
    <p>{{ user.name }}</p>
    <p>{{ user.age }}</p>
  </div>
</template>

<script>
  import { reactive } from 'vue'
  export default {
  name: 'Home',
  setup(){
    const user = reactive({
      name: 'zs',
      age: 18
    })

    setTimeout(() => {
      // 赋值
      user.age++
    }, 2000);
    
    return {
      user
    }
  }
}
</script>

# toRefs

将一个reactive对象转化为属性全部为ref对象的普通对象

<template>
  <div class="home">
  	<!--模板中使用-->  
    <p>{{ name }}</p>
    <p>{{ age }}</p>
  </div>
</template>

<script>
  import { reactive, toRefs } from 'vue'

  export default {
  name: 'Home',
  setup(){
    const user = reactive({
      name: 'zs',
      age: 18
    })

    return {
      // 使用扩展运算符解构出user属性后模板中可以直接使用
      ...toRefs(user)
    }
    
  }
}
</script>

# setup

组件入口，聚合数据操作


<script>
import { defineComponent, ref, reactive, computed, watch, onBeforeMount, onMounted } from "vue"
export default defineComponent({
  name: "HelloWorld",
  setup() {
    // 数据声明
    const count = ref(0);
    const user = reactive({
      name: "zs",
      age: 18,
    })
    // 方法
    const updateCount = () => {
      count.value++
    };
    const updateUser = () => {
      user.name = "ls"
    };
    // 计算属性
    const doubleCount = computed(() => count.value * 2)
    // 数据侦听
    watch(count, (newValue, oldValue) => {
      console.log("count新值", newValue)
      console.log("count旧值", oldValue)
    });
    // 钩子函数
    onBeforeMount(()=>{
      console.log('页面渲染完成前')
    })
    onMounted(()=>{
      console.log('页面渲染完成')
    })

    return {
      count,
      user,
      updateCount,
      updateUser,
      doubleCount,
    }
    
  }
})
</script>

# watch

数据侦听

  • 接收三个参数

    • 一个响应式引用或我们想要侦听的 getter 函数
    • 一个回调
    • 可选的配置选项

  • 侦听reactive定义的数据

    const user = reactive({
  name: 'zs',
  age: 18
})

watch(()=> user.age,(newValue,oldValue)=>{

  console.log('oldValue',oldValue);

# watchEffect

数据侦听

  • 深度侦听
  • 不需要手动传入依赖
  • 会先执行一次用来自动收集依赖
  • 无法获取到变化前的值， 只能获取变化后的值
watchEffect(()=>{
  console.log('counter',counter.value);
  console.log('user',user);
})

  get: () => counter.value * 2,
  set: (val) => counter.value = val / 2

# 生命周期

  • beforeDestroy名称变更成beforeUnmount; destroyed 表更为 unmounted
  • 增加setup
  • 增加on开头的API，其执行时机更快
<script>
import {defineComponent, onBeforeMount, onBeforeUnmount, onBeforeUpdate, onMounted, onUnmounted, onUpdated} from 'vue'
  
export default defineComponent({
  name: 'HelloWorld',
  setup(){
    console.log('setup');
    onBeforeMount(()=>console.log('onBeforeMount'))
    onMounted(()=>console.log('onMounted'))
    onBeforeUpdate(()=>console.log('onBeforeUpdate'))
    onUpdated(()=>console.log('onUpdated'))
    onBeforeUnmount(()=>console.log('onBeforeUnmount'))
    onUnmounted(()=>console.log('onUnmounted'))
  },
  beforeCreate(){
    console.log('beforeCreate');
  },
  created(){
    console.log('created');
  },
  beforeMount(){
    console.log('beforeMount');
  },
  mounted(){
    console.log('mounted');
  },
  beforeUpdate(){
    console.log('beforeUpdate');
  },
  updated(){
    console.log('updated');
  },
  beforeUnmount(){
    console.log('beforeUnmount');
  },
  unmounted(){
    console.log('unmounted');
  }
})
</script>

# Teleport

提供了一种干净的方法，允许我们控制在 DOM 中哪个父节点下呈现 HTML

app.component('modal-button', {
  template: `
    <button @click="modalOpen = true">
        Open full screen modal! (With teleport!)
    </button>

    <teleport to="body">
      <div v-if="modalOpen" class="modal">
        <div>
          I'm a teleported modal! 
          (My parent is "body")
          <button @click="modalOpen = false">
            Close
          </button>
        </div>
      </div>
    </teleport>
  `,
  data() {
    return { 
      modalOpen: false
    }
  }
})

# 自定义指令

自定义指令钩子函数更新如下

# v-model

  • 变更：在自定义组件上使用v-model时， 属性以及事件的默认名称变了
    • prop：value -> modelValue
    • event：input -> update:modelValue
  • 变更：v-bind的.sync修饰符被去掉了, 合并到了v-model里
  • 新增：同一组件可以同时设置多个 v-model
  • 新增：开发者可以自定义 v-model修饰

vue2

<search-input v-model="searchValue"><search-input>

<!-- 相当于 -->
<search-input :value="searchValue" @input="searchValue=$event"><search-input>
