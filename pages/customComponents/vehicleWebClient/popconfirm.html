# Popconfirm 确认删除对话框

# 说明

该组件用于确认删除对话框。

# 示例

# 代码示例

点击查看代码
<template>
  <popconfirm
    title="删除小组"
    content="删除小组后关联车辆所属小组全部清空，是否确认删除？"
    @onconfirm="onConfirm"
  >
    <span class="del">删除</span>
  </popconfirm>

</template>

<script>
  import { message } from 'ant-design-vue';
  import { defineComponent } from 'vue'
  export default defineComponent({
    setup() {
      /**
      * @description: 点击确认删除
      */
      function onConfirm() {
        message.info('点击确认删除');      
      }

      return {
        onConfirm
      }
    },
  })
</script>

<style lang="less" scoped>
 .del {
  color: #f35151;
  cursor: pointer;
 }
</style>
