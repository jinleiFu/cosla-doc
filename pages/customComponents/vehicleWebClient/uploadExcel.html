# UploadExcel 导入导出excel

# 说明

该组件用于导入导出excel

# 示例

# 代码示例

点击查看代码
<template>
  <div>
    <a-button type="primary" @click="showUploadExcel = true">点击显示导入弹窗</a-button>
    <upload-excel 
      v-model:visible="showUploadExcel"
      template-url="http://static.shinwell.cn/型号库导入模板.xlsx"
      :headerIndex="1"
      :columns="excelColumns"
    ></upload-excel>  
  </div>
</template>

<script>
  import { defineComponent, reactive, toRefs } from 'vue'
  export default defineComponent({
    setup(_, { emit }) {
      const excelColumns = [
        {title:'配件名称', key:'fittingName'},
        {title:'品类', key:'category'},
        {title:'品牌', key:'brand'},
        {title:'型号ID', key:'modelId'},
        {title:'型号', key:'model'}
      ]
      // 数据源
      const state = reactive({
        showUploadExcel: false,
      })

      return {
        ...toRefs(state),
        excelColumns
      }
    },
  })
</script>

Props
